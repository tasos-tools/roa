async function verifyAccess(code) { const statusEl = document.getElementById('status'); statusEl.textContent = 'üîÑ V√©rification...'; statusEl.className = 'loading'; try { // ‚úÖ FIX HASH PARSING const hashMatch = window.location.hash.match(/#(.*)/); let hashData = {}; if (hashMatch) { try { hashData = JSON.parse(atob(hashMatch[1])); } catch(e) { console.log('Hash parse failed:', e); } } const {verifier, state} = hashData; if (!verifier) throw new Error('PKCE dans URL manquant'); const urlState = new URLSearchParams(window.location.search).get('state'); if (state !== urlState) throw new Error('√âtat invalide'); // Token exchange const tokenResponse = await fetch('https://discord.com/api/oauth2/token', { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded' }, body: new URLSearchParams({ client_id: CLIENT_ID, grant_type: 'authorization_code', code: code, redirect_uri: window.location.origin + window.location.pathname + window.location.hash, code_verifier: verifier }) }); if (!tokenResponse.ok) { const errorText = await tokenResponse.text(); throw new Error(`Token √©chou√©: ${errorText}`); } const tokenData = await tokenResponse.json(); // V√©rif serveur const guildsResponse = await fetch('https://discord.com/api/users/@me/guilds', { headers: { Authorization: `Bearer ${tokenData.access_token}` } }); const guilds = await guildsResponse.json(); const inGuild = guilds.some(guild => guild.id === GUILD_ID); if (inGuild) { window.location.replace(window.location.pathname); showContent(); } else { throw new Error('Non membre du serveur'); } } catch (error) { statusEl.innerHTML = `‚ùå ${error.message}`; statusEl.className = 'error'; } }
